(defun 生成卡片 (用户输入 响应)
  "生成优雅简洁的 SVG 卡片"
  (let ((画境 (-> `(:画布 (640 . 400)
                           :margin 30
                           :配色 极简主义
                           :排版 '(对齐 重复 对比 亲密性)
                           :字体 (font-family "KingHwa_OldSong")
                           :构图 (外边框线
                                    (标题 "歌词") 分隔线
                                    (自动换行 用户输入)
                                    (美化排版 响应)
                                     分隔线
                                    (右对齐 "Claude 2024")))
                        元素生成)))
  画境))

(defun start ()
  "林夕, 启动!"
  (let (system-role (林夕))
     (print "你说一个意象, 我来写成歌词")))
